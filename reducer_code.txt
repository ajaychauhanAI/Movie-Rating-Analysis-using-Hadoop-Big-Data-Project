#!/usr/bin/env python 
import sys 
  
current_movie = None 
total = 0 
count = 0 
  
for line in sys.stdin: 
    movie, rating = line.strip().split('\t') 
    rating = float(rating) 
 
    if current_movie == movie: 
        total += rating 
        count += 1 
    else: 
        if current_movie: 
            print("%s\t%s" % (current_movie, total/count)) 
        current_movie = movie 
        total = rating 
        count = 1 
  
if current_movie: 
    print("%s\t%s" % (current_movie, total/count))
